# 工作流注册表方案

> 创建日期: 2026-05-28
> 状态: 设计阶段
> 原则: 硬编码注册、低耦合高内聚、RunningHub 默认引擎、**前后端分离**

---

## 一、现状问题

### 当前代码中工作流相关逻辑散落在多处

| 位置 | 问题 |
|------|------|
| `stores/config.ts` | `imageModel` 硬编码默认值 `'nunchaku_qwen'`（ComfyUI 本地工作流名），与 RunningHub 无关 |
| `stores/config.ts` | `imageEngine` 默认值是 `'comfyui_local'`，但用户要求默认使用 RunningHub |
| `usePipeline.ts:112-143` | 前端直接调 `comfyui.executeWorkflow` / `runninghub.createTask`，分支逻辑+节点映射硬编码在前端 |
| `config/constants.ts` | `T2I_MODELS`、`IMAGE_ENGINE_MODES` 把后端工作流配置放在了前端常量中 |
| `services/comfyui.ts` | 前端直接发 workflowId + replacements |
| `services/runninghub.ts` | 前端直接发 workflowId + nodeParams，包含后端 API 调用细节 |

### 核心矛盾

1. **工作流 ID、节点映射、引擎分支逻辑**是后端职责 — 它们决定调用哪个第三方 API（RunningHub/ComfyUI），用什么参数格式
2. 当前这些逻辑全部在前端 `usePipeline.ts` 中，违反了 code-standards 的 **前后端分离** 原则
3. 前端不应该知道 RunningHub 的 `nodeId`/`fieldName` 参数结构，也不应该知道 ComfyUI 的 `replacements` 格式

---

## 二、设计目标

1. **注册表放后端** — 工作流注册表定义在 `config.yaml`，后端通过 API 暴露给前端
2. **执行逻辑归后端** — 前端只发 `POST /api/v1/workflow/execute { workflow_id, prompt, width, height }`，后端根据注册表决定调 RH 还是 ComfyUI，以及如何拼参数
3. **前端只做 UI** — 获取工作流列表、展示、选择、发送执行请求
4. **硬编码注册** — 新增工作流 = 在 `config.yaml` 加一段配置，不需要动态扫描
5. **RunningHub 默认** — 默认激活的工作流是 RunningHub 类型

### 3.2 新建 `services/workflow_service.py`

职责：读取 config.yaml 注册表 + 统一执行入口

```python
# backend/services/workflow_service.py

"""工作流注册表服务 — 读取配置、统一执行分发。"""

from typing import Any
from core.config import get_config
from services import comfyui_service, rh_service


def get_registry() -> list[dict[str, Any]]:
    """返回所有注册的工作流。"""
    config = get_config()
    return config.get("workflows", {}).get("registry", [])


def get_enabled_workflows(category: str | None = None) -> list[dict[str, Any]]:
    """返回所有已启用的工作流（可选按类别过滤）。"""
    return [
        w for w in get_registry()
        if w.get("enabled", False) and (category is None or w.get("category") == category)
    ]


def get_workflow_by_id(workflow_id: str) -> dict[str, Any] | None:
    """根据 ID 查找工作流。"""
    for w in get_registry():
        if w.get("id") == workflow_id:
            return w
    return None


def get_default_workflow_id() -> str:
    """返回默认工作流 ID。"""
    config = get_config()
    return config.get("workflows", {}).get("default", "")


async def execute(workflow_id: str, prompt: str, width: int, height: int) -> dict[str, Any]:
    """
    统一执行入口。
    根据注册表中的 engine 类型分发到 comfyui_service 或 rh_service。
    前端不需要知道分发细节。
    """
    wf = get_workflow_by_id(workflow_id)
    if not wf:
        raise ValueError(f"工作流不存在: {workflow_id}")
    if not wf.get("enabled"):
        raise ValueError(f"工作流已禁用: {workflow_id}")

    engine = wf.get("engine", "")

    if engine == "comfyui_local":
        return await _execute_comfyui(wf, prompt, width, height)
    elif engine == "runninghub":
        return await _execute_runninghub(wf, prompt, width, height)
    else:
        raise ValueError(f"未知引擎类型: {engine}")


async def _execute_comfyui(wf: dict, prompt: str, width: int, height: int) -> dict[str, Any]:
    """ComfyUI 本地执行。"""
    comfyui_id = wf.get("comfyui_workflow_id", "")
    workflow_json = await comfyui_service.get_workflow(comfyui_id)
    if not workflow_json:
        raise ValueError(f"ComfyUI 工作流文件不存在: {comfyui_id}")
    result = await comfyui_service.execute_workflow(
        workflow_json,
        {"prompt": prompt, "width": str(width), "height": str(height)},
    )
    return {"engine": "comfyui_local", **result}


async def _execute_runninghub(wf: dict, prompt: str, width: int, height: int) -> dict[str, Any]:
    """RunningHub 云端执行。"""
    rh_id = wf.get("rh_workflow_id", "")
    mapping = wf.get("rh_node_mapping", {})

    node_params = []
    if "prompt" in mapping:
        m = mapping["prompt"]
        node_params.append({"nodeId": m["node_id"], "fieldName": m["field_name"], "fieldValue": prompt})
    if "width" in mapping:
        m = mapping["width"]
        node_params.append({"nodeId": m["node_id"], "fieldName": m["field_name"], "fieldValue": str(width)})
    if "height" in mapping:
        m = mapping["height"]
        node_params.append({"nodeId": m["node_id"], "fieldName": m["field_name"], "fieldValue": str(height)})

    result = await rh_service.create_task(rh_id, node_params)
    return {"engine": "runninghub", **result}
```

### 3.3 新建 `routers/workflow.py`

薄路由层，暴露 3 个 API：

```python
# backend/routers/workflow.py

"""工作流路由 — 列表/详情/执行。"""

from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
from services import workflow_service

router = APIRouter(prefix="/api/v1/workflow", tags=["Workflow"])


@router.get("/list")
async def list_workflows(category: str | None = None):
    """列出已启用的工作流（前端用于展示列表）。"""
    workflows = workflow_service.get_enabled_workflows(category)
    default_id = workflow_service.get_default_workflow_id()
    # 只返回前端需要的字段，不暴露 node_mapping 等后端细节
    items = [
        {
            "id": w["id"],
            "name": w["name"],
            "category": w.get("category", ""),
            "engine": w.get("engine", ""),
            "default_size": w.get("default_size", ""),
            "description": w.get("description", ""),
        }
        for w in workflows
    ]
    return {"workflows": items, "default": default_id}


@router.get("/active")
async def get_active():
    """返回默认/推荐的工作流 ID。"""
    return {"workflow_id": workflow_service.get_default_workflow_id()}


class ExecuteRequest(BaseModel):
    workflow_id: str
    prompt: str
    width: int = 720
    height: int = 1280


@router.post("/execute")
async def execute(req: ExecuteRequest):
    """统一执行入口 — 前端只传 workflow_id + prompt + 尺寸。"""
    try:
        result = await workflow_service.execute(req.workflow_id, req.prompt, req.width, req.height)
        return result
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))
    except RuntimeError as e:
        raise HTTPException(status_code=502, detail=str(e))
```

### 4.1 新建 `services/workflow.ts`

前端通过 API 获取列表和执行，不包含任何后端逻辑：

```ts
// frontend/src/services/workflow.ts

import api from './api'

/** 工作流列表项（API 返回结构） */
export interface WorkflowItem {
  id: string
  name: string
  category: string
  engine: string
  default_size: string
  description: string
}

/** 获取已启用的工作流列表 */
export async function listWorkflows(category?: string): Promise<{ workflows: WorkflowItem[]; default: string }> {
  const resp = await api.get('/workflow/list', { params: category ? { category } : {} })
  return resp.data
}

/** 执行工作流（统一入口） */
export async function executeWorkflow(workflowId: string, prompt: string, width: number, height: number) {
  const resp = await api.post('/workflow/execute', { workflow_id: workflowId, prompt, width, height })
  return resp.data
}
```

## 五、数据流总览

```
config.yaml workflows.registry（后端硬编码，唯一真相来源）
        │
        ▼
workflow_service.py（读注册表 + 执行分发）
        │
        ├── GET  /api/v1/workflow/list     → 前端展示列表
        ├── GET  /api/v1/workflow/active    → 前端获取默认 ID
        └── POST /api/v1/workflow/execute   → 统一执行
                │
                ├── engine=comfyui_local → comfyui_service.execute_workflow()
                └── engine=runninghub   → rh_service.create_task()

前端:
  WorkflowManager.vue  ─── GET /workflow/list ──→ 展示卡片列表
        │                                          │
        ▼                                          ▼
  configStore.activeWorkflowId               点击切换选中
        │
        ▼
  usePipeline.ts runStep2 ─── POST /workflow/execute ──→ 后端分发
```

**关键区别**: 前端完全不知道 RunningHub 节点映射、ComfyUI replacements 格式等后端细节。
